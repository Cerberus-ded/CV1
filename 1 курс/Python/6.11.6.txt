name=('Картофель', 'Укроп', 'Морковь', 'Горох', 'Капуста', 'Редис')
a=set(name)
b=('Картофель', 'Морковь', 'Редис','Горох')
a1=set(b)
c=('Картофель','Укроп','Морковь','Горох')
a2=set(c)
d=('Картофель', 'Морковь', 'Укроп','Редис')
a3=set(d)
print('Не встречается:')
print(a.difference(a1,a2,a3))
print()
print('Встречаются хотя бы один раз: ')
print(a-a.difference(a1,a2,a3))
print()
print('Встречается во всех классах:')
print(a.intersection(a1,a2,a3))
